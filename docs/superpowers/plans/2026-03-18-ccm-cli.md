# CCM CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Refactor the claude-context-manager repo from a "template you clone" into a proper CLI tool (`ccm`) users install once and use in any project.

**Architecture:** Separate the tool (bin/ccm, install.sh, reusable GitHub Action) from the templates (what ccm init generates). A unified `ccm` bash CLI with subcommands `init` and `add` handles all user-facing setup. A global `~/.ccm/config` and a per-project `.ccm` file bridge meta-repo and sub-repos.

**Tech Stack:** Bash, jq, git, optional gh (GitHub CLI). bats-core for tests.

```bash
# lib/config.sh
CCM_HOME="${CCM_HOME:-$HOME/.ccm}"
CCM_GLOBAL_CONFIG="$CCM_HOME/config"

get_meta_repo() {
  if [[ ! -f "$CCM_GLOBAL_CONFIG" ]]; then
    echo ""
    return 1
  fi
  jq -r '.default_meta' "$CCM_GLOBAL_CONFIG"
}

set_meta_repo() {
  local path="$1"
  mkdir -p "$CCM_HOME"
  if [[ -f "$CCM_GLOBAL_CONFIG" ]]; then
    jq --arg p "$path" '.default_meta = $p' "$CCM_GLOBAL_CONFIG" \
      > "$CCM_GLOBAL_CONFIG.tmp" && mv "$CCM_GLOBAL_CONFIG.tmp" "$CCM_GLOBAL_CONFIG"
  else
    echo "{\"default_meta\": \"$path\"}" > "$CCM_GLOBAL_CONFIG"
  fi
}
```

## Task 5: Create the reusable GitHub Action

**Files:**
- Create: `.github/workflows/sync.yml`

- [ ] **Step 1: Create `.github/workflows/sync.yml`**

The reusable workflow lives in the CCM repo. It checks out the meta-repo and uses `lib/merge.sh` from that same checkout — no network fetches at runtime:

```yaml
name: CCM Sync (Reusable)

on:
  workflow_call:

jobs:
  sync:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout meta-repo
        uses: actions/checkout@v4
        with:
          token: ${{ secrets.GITHUB_TOKEN }}

      - name: Checkout CCM tools
        uses: actions/checkout@v4
        with:
          repository: yves-s/claude-context-manager
          path: .ccm-tools

      - name: Install jq
        run: sudo apt-get install -y jq

      - name: Sync CLAUDE.md to sub-repos
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          set -euo pipefail
          MERGE_SH=".ccm-tools/lib/merge.sh"
          SUB_COUNT=$(jq '.sub_projects | length' ccm.json)
          if [[ "$SUB_COUNT" == "0" ]]; then
            echo "No sub-projects registered."
            exit 0
          fi
          for i in $(seq 0 $((SUB_COUNT - 1))); do
            REPO=$(jq -r ".sub_projects[$i].repo" ccm.json)
            MARKER=$(jq -r ".sub_projects[$i].marker" ccm.json)
            REPO_NAME=$(basename "$REPO" .git)
            echo "→ Syncing $REPO_NAME"
            git clone --depth 1 \
              "https://x-access-token:${GH_TOKEN}@${REPO#git@github.com:}" \
              "/tmp/$REPO_NAME" 2>/dev/null \
              || { echo "  ⚠ Could not clone $REPO — skipping"; continue; }
            bash "$MERGE_SH" CLAUDE.md "/tmp/$REPO_NAME/CLAUDE.md" "$MARKER"
            cd "/tmp/$REPO_NAME"
            if git diff --quiet CLAUDE.md; then
              echo "  ✅ Already up to date"
            else
              git config user.email "ccm-bot@users.noreply.github.com"
              git config user.name "CCM Sync Bot"
              git add CLAUDE.md
              git commit -m "chore: sync CLAUDE.md from meta-repo [skip ci]"
              git push
              echo "  ✅ Synced"
            fi
            cd -
          done
```

```bash
#!/bin/bash
set -euo pipefail

CCM_VERSION="0.1.0"
CCM_REPO="${CCM_REPO:-yves-s/claude-context-manager}"

# Path resolution: installed mode (~/.ccm/) vs development mode (repo)
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
if [[ -f "$SCRIPT_DIR/../lib/config.sh" ]]; then
  # Development mode: running from repo bin/
  LIB_DIR="$(dirname "$SCRIPT_DIR")/lib"
  TEMPLATES_DIR="$(dirname "$SCRIPT_DIR")/templates"
else
  # Installed mode: binary in /usr/local/bin, assets in ~/.ccm/
  LIB_DIR="$HOME/.ccm/lib"
  TEMPLATES_DIR="$HOME/.ccm/templates"
fi

CCM_HOME="${CCM_HOME:-$HOME/.ccm}"
# shellcheck source=../lib/config.sh
source "$LIB_DIR/config.sh"

cmd_init() {
  if [[ "${CCM_NON_INTERACTIVE:-}" == "1" ]]; then
    GITHUB_USER="$CCM_GITHUB_USER"
    PROJECT_NAME="$CCM_PROJECT_NAME"
    NOTION_ENABLED="${CCM_NOTION:-false}"
  else
    echo "🚀 CCM — Meta-Repo Setup"
    echo "========================"
    read -rp "GitHub Username oder Organisation: " GITHUB_USER
    [[ -z "$GITHUB_USER" ]] && { echo "❌ Erforderlich."; exit 1; }
    read -rp "Projektname: " PROJECT_NAME
    PROJECT_NAME="${PROJECT_NAME:-Mein Projekt}"
    read -rp "Notion-Integration aktivieren? (y/n): " _NOTION
    [[ "$_NOTION" == "y" ]] && NOTION_ENABLED=true || NOTION_ENABLED=false
  fi

  REPO_NAME=$(basename "$(pwd)")
  META_REPO_URL="git@github.com:$GITHUB_USER/$REPO_NAME.git"

  # context/ — only copy if not already present (idempotency)
  if [[ ! -d "context" ]]; then
    cp -r "$TEMPLATES_DIR/meta/context" .
  fi

  # CLAUDE.md — always regenerate from template (contains meta config)
  sed \
    -e "s|{{PROJECT_NAME}}|$PROJECT_NAME|g" \
    -e "s|{{GITHUB_USER}}|$GITHUB_USER|g" \
    -e "s|{{META_REPO_URL}}|$META_REPO_URL|g" \
    "$TEMPLATES_DIR/meta/CLAUDE.md" > CLAUDE.md

  # ccm.json — only create if not already present
  if [[ ! -f "ccm.json" ]]; then
    sed \
      -e "s|{{PROJECT_NAME}}|$PROJECT_NAME|g" \
      -e "s|{{GITHUB_USER}}|$GITHUB_USER|g" \
      -e "s|{{META_REPO_URL}}|$META_REPO_URL|g" \
      -e "s|{{NOTION_ENABLED}}|$NOTION_ENABLED|g" \
      "$TEMPLATES_DIR/meta/ccm.json" > ccm.json
  fi

  # GitHub Action workflow
  mkdir -p .github/workflows
  sed -e "s|{{CCM_REPO}}|$CCM_REPO|g" \
    "$TEMPLATES_DIR/meta/github-workflow.yml" > .github/workflows/sync.yml

  # Register in global config
  set_meta_repo "$(pwd)"

  echo ""
  echo "✅ Meta-Repo konfiguriert"
  echo "   Projekt:  $PROJECT_NAME"
  echo "   GitHub:   $GITHUB_USER"
  echo ""
  echo "Nächste Schritte:"
  echo "  1. context/ Dateien anpassen"
  echo "  2. git remote add origin $META_REPO_URL"
  echo "  3. git push -u origin main"
  echo "  4. Sub-Projekt hinzufügen: ccm add (im Sub-Repo)"
}

cmd_add() {
  # defined in Task 7 — placeholder to satisfy router
  echo "cmd_add not yet implemented" && exit 1
}

case "${1:-}" in
  init)    cmd_init ;;
  add)     cmd_add ;;
  version) echo "ccm $CCM_VERSION" ;;
  *)
    echo "Usage: ccm <command>"
    echo ""
    echo "Commands:"
    echo "  init     Meta-Repo im aktuellen Verzeichnis einrichten"
    echo "  add      Dieses Projekt als Sub-Repo registrieren"
    echo "  version  Version anzeigen"
    ;;
esac
```

Replace the `cmd_add` placeholder with the full implementation. On re-run, if `CLAUDE.md` already exists, use `merge.sh` to update only the meta section while preserving project content:

```bash
cmd_add() {
  META_PATH=$(get_meta_repo)
  if [[ -z "$META_PATH" || ! -d "$META_PATH" ]]; then
    echo "❌ Kein Meta-Repo gefunden."
    echo "   Führe zuerst 'ccm init' in deinem Meta-Repo aus."
    exit 1
  fi

  META_CONFIG="$META_PATH/ccm.json"
  META_REPO_URL=$(jq -r '.meta_repo' "$META_CONFIG")

  if [[ "${CCM_NON_INTERACTIVE:-}" == "1" ]]; then
    REPO_NAME="$CCM_REPO_NAME"
    REPO_URL="$CCM_REPO_URL"
  else
    REPO_NAME=$(basename "$(pwd)")
    GITHUB_USER=$(jq -r '.github_user' "$META_CONFIG")
    REPO_URL="git@github.com:$GITHUB_USER/$REPO_NAME.git"
    echo "🔗 Sub-Projekt registrieren"
    echo "   Meta-Repo: $META_PATH"
    read -rp "   Repo-URL [$REPO_URL]: " _URL
    REPO_URL="${_URL:-$REPO_URL}"
  fi

  MARKER="<!-- PROJECT CONTEXT BELOW -->"

  # Register in meta-repo (idempotent)
  EXISTING=$(jq -r --arg url "$REPO_URL" \
    '.sub_projects[] | select(.repo == $url) | .repo' "$META_CONFIG")
  if [[ -z "$EXISTING" ]]; then
    jq --arg repo "$REPO_URL" --arg marker "$MARKER" \
      '.sub_projects += [{"repo": $repo, "marker": $marker}]' \
      "$META_CONFIG" > "$META_CONFIG.tmp" && mv "$META_CONFIG.tmp" "$META_CONFIG"
  fi

  # Create or update CLAUDE.md
  if [[ -f "CLAUDE.md" ]]; then
    # Preserve project section — only update meta part
    bash "$LIB_DIR/merge.sh" "$META_PATH/CLAUDE.md" "CLAUDE.md" "$MARKER"
  else
    # First run — assemble from templates
    {
      cat "$META_PATH/CLAUDE.md"
      echo ""
      echo "---"
      echo ""
      sed "s|{{REPO_NAME}}|$REPO_NAME|g; s|{{REPO_URL}}|$REPO_URL|g" \
        "$TEMPLATES_DIR/sub/CLAUDE.md"
    } > CLAUDE.md
  fi

  # Create .ccm marker (idempotent overwrite is fine — it's generated content)
  sed "s|{{META_REPO_URL}}|$META_REPO_URL|g" \
    "$TEMPLATES_DIR/sub/dot-ccm" > .ccm

  echo ""
  echo "✅ $REPO_NAME registriert"
  echo "   CLAUDE.md erstellt/aktualisiert"
  echo "   .ccm Marker erstellt"
  echo ""
  echo "Nächste Schritte:"
  echo "  1. CLAUDE.md Projekt-Kontext anpassen (unterhalb des Markers)"
  echo "  2. git add CLAUDE.md .ccm && git commit -m 'chore: add CCM context'"
  echo "  3. git push → Meta-Repo Action synct künftig automatisch"
}
```
